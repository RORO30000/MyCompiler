funcion calcularPotencia(entero base, entero exponente) retornar entero {
    entero resultado = 1;
    entero i = 0;
    mientras (i < exponente) {
        resultado = resultado * base;
        i = i + 1;
    }
    fin_mientras
    retornar resultado;
}

funcion calcularCuadrado(entero n) retornar entero {
    retornar n * n;
}
